/**
 * @module Customer Credit Model
 */

var customerCredit = require("../models/customerCreditModel.js");

module.exports = {
  getCredit: function(req, handler) {
    customerCredit.getCredits(req.body.customerId, handler);
  },
  // overwrites original credit value with new credit value
  replaceCreditValue: function(req, handler) {
    customerCredit.replaceCreditValue(req.body.customerId,
      req.body.credits, handler);
  },
  // changes credit value by number of credits specified
  changeCreditValue: function(req, handler) {
    customerCredit.modifyCreditValue(req.body.customerId,
      req.body.credits, handler);
  }
};

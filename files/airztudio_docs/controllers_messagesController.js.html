/**
 * @summary Messages controller
 * @description The messages controller handles received post requests for messages
 * and routes the request to the appropriate messaging model functions
 * @module messagesController
 * @requires messagesModel
 * @author hautonjt
 */

var messagesModel = require("../models/messagesModel.js");

module.exports = {
  sendMessage: function(req, handler) {
    messagesModel.sendMessage(req.session.msgUserId, req.body.recipient,
      req.body.message, handler);
  },
  readMessages: function(req, handler) {
    messagesModel.readMessages(req.session.msgUserId, req.body.startTime,
      req.body.endTime, handler);
  },
  getMessageByID: function(req, handler) {
    messagesModel.getMessageByID(req.body.messageId, handler);
  },
  markMessageAsRead: function(req, handler) {
    messagesModel.markMessageAsRead(req.body.sender,
      req.body.recipient, handler);
  },
  recipientMarkMessageAsDeleted: function(req, handler) {
    messagesModel.recipientMarkMessageAsDeleted(req.body.messageId, handler);
  },
  senderMarkMessageAsDeleted: function(req, handler) {
    messagesModel.senderMarkMessageAsDeleted(req.body.messageId, handler);
  },
  deleteMessage: function(req, handler) {
    messagesModel.deleteMessage(req.body.messageId, handler);
  }

};

/**
 * @module packageRoutes
 */
var packageController =
  require("../controllers/studioPackagesController.js");
var studioAccountController =
  require("../controllers/studioAccountController.js");
var customerAccountController =
  require("../controllers/customerAccountController.js");

module.exports = {
    init : function(app) {
      app.get("/packageView", function(req, res) {
        res.render("pages/packageList");
      });

      app.get("/packageView/:package_id", function (req, res) {
        packageController.get({
          "body": { "packageId": req.params.package_id }
        }, res, function(data){
          res.render("pages/packageView", {"result": data.result});  //TODO
        });

      });

      app.get("/packageSearch", function(req, res) {
        res.render("pages/packageSearch");
      });

      app.post("/packageSearch", function(req, res){
        packageController.search(req, res, function(data){
          res.send(data);
        });
      });

      /**
       * This route appends to the shoppingCart object in sess
       */
      app.post("/bookPackage", function(req, res) {
        if (req.session.acct === "customer") {
          console.log("Request: " + JSON.stringify(req.body));
          //if (req.session.shoppingCart == null) {
          req.session.shoppingCart = [req.body];
          //} else {
          //  req.session.shoppingCart.push(req.body);
          //}
          packageController.createBooking(req, res, function(result) {
            res.send(result);
          });
        } else {
          res.redirect("/");
        }
      });

      app.get("/createPackage", function(req, res) {
        if (req.session.acct === "studio") {
          res.render("pages/createPackage");
        } else {
          res.redirect("/");
        }
      });

      app.get("/listPackages", function (req, res) {
        if(req.session.acct === "studio") {
          packageController.get({
            "body": { "studioId": req.session.ID }
          }, res, function(result) {
            res.render("pages/packageList", { "result": result.result });
          });
        }
        else {
          res.redirect("/");
        }
      });

      app.get("/listBookings", function (req, res) {
        if(req.session.acct === "customer") {
          packageController.getBooking({
            "body": { "customerId": req.session.ID }
          }, res, function(result) {
            res.render("pages/packageList", { "result": result.result });
          });
        }
        else {
          res.redirect("/");
        }
      });
    }
}

/**
 * @module profileRoutes
 */
var customerAccountController =
  require("../controllers/customerAccountController.js");
var studioAccountController =
  require("../controllers/studioAccountController.js");
var customerCreditController =
  require("../controllers/customerCreditController.js");

module.exports = {
  init : function(app) {
    app.get("/profileSetup", function(req, res) {
      sess = req.session;
      if(sess.acct === "customer") {
        customerAccountController.getProfile(sess.username, function(data) {
          res.render("pages/customerProfileSetup", {
            "sess": sess,
            "profileData": data.result
          });
        });
      } else if (sess.acct === "studio") {
        studioAccountController.getProfile(sess.username, function(data) {
          console.log(data.result);
          res.render("pages/studioProfileSetup", {
            "sess": sess,
            "profileData": data.result
          });
        });
      } else {
        res.redirect("/");
      }
    });

    app.get("/studioProfileView", function(req, res){
      console.log("studio query: \n"+req.query.studio);
      if(req.query.studio === "undefined") {
        res.redirect("/404");
      } else {
        studioAccountController.view(req.query.studio, req, res);
      }
    });

    app.post("/profile", function(req, res){
      //util.log(util.inspect(req.body));
      req.body.username = req.session.username;
      if(req.session.acct === "customer") {
        customerAccountController.modifyProfile(req, function(data) {
          res.send(data);
        });
      } else {
        studioAccountController.modifyProfile(req, function(data) {
          res.send(data);
        });
      }
    });
  }
};

/**
 * @summary Studio Packages controller
 * @description The studio packages controller handles received post requests
 * for studio packages and routes the request to the appropriate studio
 * package model functions.
 * @module studioPackagesController
 * @requires studioPackagesModel
 * @author hautonjt
 */

var studioPackages = require("../models/studioPackagesModel.js");

module.exports = {
  test: function(req, res, handler) {
    switch (req.body.request) {
      case "create":
        this.create(req, res, handler);
        break;
      case "modify":
        this.modify(req, res, handler);
        break;
      case "get":
        this.get(req, res, handler);
        break;
      case "delete":
        this.delete(req, res, handler);
        break;
      default:
        res.send("Request type mismatch for package.");
        break;
    }
  },

  create: function(req, res, handler) {
    studioPackages.createPackage(req.body.studioId,
      req.body.packageContents, req.body.packagePhoto, handler);
  },

  modify: function(req, res, handler) {
    studioPackages.modifyPackage(req.body.studioId,
      req.body.packageId, req.body.packageContents, handler);
  },

  get: function(req, res, handler) {
    studioPackages.getPackage(req.body.studioId,
      req.body.packageId, handler);
  },

  search: function(req, res, handler) {
    /** this function will send a request to model function search to find
    * all packages with day equal to given day
    * studioPackages.searchByDay(day_as_string, handler);
    **/
    //studioPackages.searchByDay(req.body.searchStartDate, handler);
    studioPackages.searchByCriteria(req.body.searchStartDate, req.body.searchEndDate,
      req.body.searchTag, req.body.searchStudioName,
      req.body.searchPackageName, handler);
  },

  delete: function(req, res, handler) {
    studioPackages.deletePackage(req.body.studioId, req.body.packageId,
      handler);
  },


  createBooking: function(req, res, handler) {
    studioPackages.createBooking(req.session.ID, req.session.shoppingCart,
      handler);
  },

  confirmBooking: function(req, res, handler) {
    studioPackages.confirmBooking(req.session.ID, req.body.packages,
      handler);
  },

  getBooking: function(req, res, handler) {
    studioPackages.getBooking(req.body.customerId, handler);
  }
};

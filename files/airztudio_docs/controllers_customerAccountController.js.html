/**
 * @summary Customer Account controller
 * @description The customer account controller handles received post requests
 * for customer accounts and routes the request to the appropriate customer
 * account model functions.
 * @module customerAccountController
 * @requires customerAccountModel
 */

var customerAccount = require("../models/customerAccountModel.js");

/* jshint esnext: true */

/**
 * @summary The number of millseconds in 30 days, used as the default age of a
 * cookie once logged in.
 * @constant
 * @default
 */
const COOKIE_AGE_ONCE_LOGGED_IN = 30 * 24 * 60 * 60 * 1000;

module.exports = {
  test: function(req, handler) {
    switch(req.body.request) {
      case "create":
        this.create(req, handler);
        break;
      case "delete":
        this.delete(req, handler);
        break;
      case "modify":
        this.modify(req, handler);
        break;
      case "access":
        this.getAccount(req, handler);
        break;
      case "selectAll":
        this.selectAll(handler);
        break;
      default:
        handler({
          "status": "FAIL: invalid request type",
          "result": null
        });
        break;
    }
  },
  create: function(req, handler) {
      customerAccount.createAccount(req.body, handler);
  },
  delete: function(req, handler) {
    customerAccount.deleteAccount(req.body.username, req.body.token, handler);
  },
  completeDelete: function(req, handler) {
    customerAccount.deleteAccountAndProfile(req.body.username,
      req.body.token, handler);
  },
  modify: function(req, handler) {
    customerAccount.modifyAccount(req.body.username, req.body.token,
      req.body.fieldType, req.body.newField, handler);
  },
  getAccount: function(req, handler) {
    customerAccount.accessAccount(req.body.username,
      req.body.password, handler);
  },
  getProfile: function(username, handler) {
    customerAccount.accessProfile(username, handler);
  },
  modifyProfile: function(req, handler) {
    customerAccount.modifyProfile(req.body, handler);
  },
  selectAll: function(handler) {
    customerAccount.selectAllAccounts(handler);
  },
  genAccountToken: function(req,handler) {
    customerAccount.genAccountToken(req.body.username, req.body.email,
      req.body.minWaitTime, handler);
  }
};
